# 通用配置项

```ts
export type CodegenOptions = {
  // 全局模式默认 styled
  mode: CodegenMode
  // 组件配置
  components: ComponentsOptions
  // 全局配置
  global: GlobalOptions
  // base 配置
  base: BaseOptions
  // css变量前缀，防止 css 变量冲突，默认为 `ice-`
  varPrefix: string
  // 打印日志，默认为 true
  log: boolean
  // 组件前缀，比如 `.btn` 这里设置 `ice-` 后，相关的会变成 `.ice-btn`
  // 你无需在 components 里手动添加
  prefix: string | PrefixerOptions
  // 加载的预设
  presets: DeepPartial<CodegenOptions>[]
  // 导出文件夹目录
  outdir: string
  // 不生成文件，仅仅获取 js 对象
  dryRun: boolean
  // 自定义的 tailwindcss config, 用于 apply 的 resolve
  tailwindcssConfig: TailwindcssConfig
}
```
