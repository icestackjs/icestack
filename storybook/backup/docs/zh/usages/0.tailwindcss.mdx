```js
const path = require('node:path')
const { icestackPlugin } = require('@icestack/ui/tailwindcss')

/** @type {import('tailwindcss').Config} */
module.exports = {
  // ...
  plugins: [
    icestackPlugin({
      outdir: path.resolve(__dirname, 'ice-ui')
    })
  ]
}
```
