
</Tab>
  <Tab>
  
```bash
npm install -D @icestack/preset-weapp
```

</Tab>
  <Tab>

```bash
yarn add -D @icestack/preset-weapp
```

</Tab>
  <Tab>

```bash
bun add -D @icestack/preset-weapp
```

</Tab>
</Tabs>

```js filename="icestack.config.js"
const miniprogramPreset = require('@icestack/preset-weapp')

/**
 * @type {import('@icestack/ui').Config}
 */
const config = {
  outdir: './my-ui',
  presets: [miniprogramPreset()]
}

module.exports = config
```
